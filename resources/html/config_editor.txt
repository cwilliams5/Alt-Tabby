<script>
// Theme switching - called by AHK after page loads
function setTheme(theme) {
  document.documentElement.dataset.theme = theme;
}

// Config data - injected by AHK via initSettings()
let configRegistry = [];
let configValues = {};
let originalValues = {};

// Parsed sections for rendering
let sections = [];
let activeSection = '';
let changeCount = 0;

// Initialize settings - called by AHK after page loads
window.initSettings = function(registry, values) {
  configRegistry = registry;
  configValues = values;
  originalValues = JSON.parse(JSON.stringify(values));

  // Parse registry into structured sections
  sections = [];
  let currentSection = null;
  let currentSubsection = null;

  for (const entry of configRegistry) {
    if (entry.type === 'section') {
      currentSection = { ...entry, subsections: [], settings: [] };
      currentSubsection = null;
      sections.push(currentSection);
    } else if (entry.type === 'subsection') {
      currentSubsection = { ...entry, settings: [] };
      if (currentSection) currentSection.subsections.push(currentSubsection);
    } else if (entry.s && entry.g) {
      const setting = { ...entry };
      if (currentSubsection && currentSubsection.section === entry.s) {
        currentSubsection.settings.push(setting);
      } else if (currentSection && currentSection.name === entry.s) {
        currentSection.settings.push(setting);
      }
    }
  }

  activeSection = sections[0]?.name || '';
  renderSidebar();
  renderContent();
  updateChangeCount();
};

// Format value for display (hex when fmt field says so)
function formatValue(val, type, fmt) {
  if (type === 'int' && typeof val === 'number' && fmt === 'hex') {
    return '0x' + val.toString(16).toUpperCase();
  }
  return val;
}

// Parse value from input
function parseValue(val, type) {
  if (type === 'int') {
    if (typeof val === 'string' && val.toLowerCase().startsWith('0x')) {
      return parseInt(val, 16) || 0;
    }
    return parseInt(val) || 0;
  }
  if (type === 'float') return parseFloat(val) || 0;
  return val;
}

// Render sidebar
function renderSidebar() {
  const sidebar = document.getElementById('sidebar');
  sidebar.innerHTML = sections.map(s => `
    <div class="sidebar-item ${s.name === activeSection ? 'active' : ''}"
         data-section="${s.name}" onclick="switchSection('${s.name}')">
      ${s.desc || s.name}
      <div class="item-desc">${s.subsections.length ? s.subsections.length + ' groups' : ''}</div>
    </div>
  `).join('') + '<div class="sidebar-search-label" id="sidebarSearchLabel"></div>';
}

// Create control HTML
function controlHTML(setting) {
  const id = `setting-${setting.s}-${setting.k}`;
  const value = configValues[setting.g];
  const hasRange = setting.min !== undefined && setting.max !== undefined;
  const isHex = setting.fmt === 'hex';

  if (setting.t === 'bool') {
    return `<label class="toggle">
      <input type="checkbox" id="${id}" ${value ? 'checked' : ''}
             onchange="updateValue('${setting.g}', this.checked, '${setting.t}')">
      <span class="toggle-slider"></span>
    </label>`;
  }
  if (setting.t === 'enum') {
    const opts = (setting.options || []).map(o =>
      `<option value="${o}" ${value === o ? 'selected' : ''}>${o}</option>`
    ).join('');
    return `<select id="${id}" onchange="updateValue('${setting.g}', this.value, '${setting.t}')">${opts}</select>`;
  }
  if (setting.t === 'int' && hasRange && !isHex) {
    // Slider + number input dual control
    return `<div class="range-control">
      <input type="range" id="${id}-slider" min="${setting.min}" max="${setting.max}" value="${value}"
             oninput="syncSlider('${setting.g}', '${id}', this.value, '${setting.t}')">
      <input type="number" id="${id}" min="${setting.min}" max="${setting.max}" value="${value}"
             onblur="clampAndUpdate('${setting.g}', '${id}', '${setting.t}', ${setting.min}, ${setting.max})">
      <span class="range-label">${setting.min} - ${setting.max}</span>
    </div>`;
  }
  if (setting.t === 'float' && hasRange) {
    // Slider + number input for floats
    const step = (setting.max - setting.min) <= 1 ? 0.01 : 0.1;
    return `<div class="range-control">
      <input type="range" id="${id}-slider" min="${setting.min}" max="${setting.max}" step="${step}" value="${value}"
             oninput="syncSlider('${setting.g}', '${id}', this.value, '${setting.t}')">
      <input type="number" id="${id}" min="${setting.min}" max="${setting.max}" step="${step}" value="${value}"
             onblur="clampAndUpdate('${setting.g}', '${id}', '${setting.t}', ${setting.min}, ${setting.max})">
      <span class="range-label">${setting.min} - ${setting.max}</span>
    </div>`;
  }
  if (setting.t === 'int' && isHex) {
    // Hex input with range label
    const rangeLabel = hasRange ? `<span class="range-label">0x${setting.min.toString(16).toUpperCase()} - 0x${setting.max.toString(16).toUpperCase()}</span>` : '';
    return `<div class="hex-control">
      <input type="text" id="${id}" value="${formatValue(value, 'int', 'hex')}"
             onblur="${hasRange ? `clampAndUpdate('${setting.g}', '${id}', '${setting.t}', ${setting.min}, ${setting.max})` : `updateValue('${setting.g}', this.value, '${setting.t}')`}">
      ${rangeLabel}
    </div>`;
  }
  if (setting.t === 'int') {
    return `<input type="text" id="${id}" value="${formatValue(value, 'int', setting.fmt)}"
            onchange="updateValue('${setting.g}', this.value, '${setting.t}')">`;
  }
  if (setting.t === 'float') {
    return `<input type="number" id="${id}" value="${value}"
            onchange="updateValue('${setting.g}', this.value, '${setting.t}')"
            step="0.01">`;
  }
  return `<input type="text" id="${id}" value="${value ?? ''}"
          onchange="updateValue('${setting.g}', this.value, '${setting.t}')">`;
}

// Render settings for a group
function renderSettings(settings) {
  return settings.map(s => `
    <div class="setting" data-key="${s.k}" data-global="${s.g}" data-search="${(s.d + ' ' + s.k + ' ' + s.g).toLowerCase()}">
      <div class="setting-info">
        <div class="setting-label">${s.k}</div>
        <div class="setting-desc">${s.d || ''}</div>
      </div>
      <div class="setting-control">${controlHTML(s)}</div>
    </div>
  `).join('');
}

// Render content
function renderContent() {
  const content = document.getElementById('content');
  content.innerHTML = sections.map(s => {
    let html = `<div class="section ${s.name === activeSection ? 'active' : ''}" data-section="${s.name}">`;
    html += `<div class="section-title">${s.desc || s.name}</div>`;
    if (s.long) html += `<div class="section-desc">${s.long}</div>`;

    // Top-level settings (before any subsection)
    if (s.settings.length) {
      html += renderSettings(s.settings);
    }

    // Subsections
    for (const sub of s.subsections) {
      html += `<div class="subsection" data-search="${(sub.name + ' ' + (sub.desc||'')).toLowerCase()}">`;
      html += `<div class="subsection-title">${sub.name}</div>`;
      if (sub.desc) html += `<div class="subsection-desc">${sub.desc}</div>`;
      html += renderSettings(sub.settings);
      html += `</div>`;
    }

    html += `</div>`;
    return html;
  }).join('');
}

// Switch section
function switchSection(name) {
  activeSection = name;
  document.querySelectorAll('.sidebar-item').forEach(el => {
    el.classList.toggle('active', el.dataset.section === name);
  });
  document.querySelectorAll('.section').forEach(el => {
    el.classList.toggle('active', el.dataset.section === name);
  });
  document.getElementById('content').scrollTop = 0;
}

// Update value
function updateValue(globalName, value, type) {
  configValues[globalName] = parseValue(value, type);
  updateChangeCount();
}

// Sync slider -> number input
function syncSlider(globalName, baseId, sliderValue, type) {
  const textInput = document.getElementById(baseId);
  if (textInput) textInput.value = sliderValue;
  updateValue(globalName, sliderValue, type);
}

// Clamp number input value and sync slider
function clampAndUpdate(globalName, baseId, type, min, max) {
  const textInput = document.getElementById(baseId);
  if (!textInput) return;
  let val = parseValue(textInput.value, type);
  if (val < min) val = min;
  if (val > max) val = max;
  // Preserve hex display for hex values
  if (type === 'int' && String(textInput.value).toLowerCase().startsWith('0x')) {
    textInput.value = '0x' + val.toString(16).toUpperCase();
  } else {
    textInput.value = val;
  }
  const slider = document.getElementById(baseId + '-slider');
  if (slider) slider.value = val;
  updateValue(globalName, val, type);
}

function updateChangeCount() {
  changeCount = 0;
  for (const key in configValues) {
    if (configValues[key] !== originalValues[key]) changeCount++;
  }
  const info = document.getElementById('changeInfo');
  const saveBtn = document.getElementById('saveBtn');
  if (changeCount > 0) {
    info.innerHTML = `<span class="change-count">${changeCount}</span> unsaved change${changeCount > 1 ? 's' : ''}`;
    saveBtn.disabled = false;
  } else {
    info.textContent = '';
    saveBtn.disabled = true;
  }
}

function getModifiedValues() {
  const changes = {};
  for (const key in configValues) {
    if (configValues[key] !== originalValues[key]) {
      changes[key] = configValues[key];
    }
  }
  return changes;
}

function saveChanges() {
  const changes = getModifiedValues();
  window.chrome.webview.postMessage(JSON.stringify({action: 'save', changes: changes}));
}

function cancelEditor() {
  window.chrome.webview.postMessage(JSON.stringify({action: 'cancel'}));
}

function resetAll() {
  if (!confirm('Reset all settings to defaults?')) return;

  // Reset to defaults from registry
  for (const entry of configRegistry) {
    if (entry.g && entry.hasOwnProperty('default')) {
      configValues[entry.g] = entry.default;
      const id = `setting-${entry.s}-${entry.k}`;
      const el = document.getElementById(id);
      if (!el) continue;
      if (entry.t === 'bool') el.checked = entry.default;
      else el.value = formatValue(entry.default, entry.t, entry.fmt);
      // Sync slider if present
      const slider = document.getElementById(id + '-slider');
      if (slider) slider.value = entry.default;
    }
  }
  updateChangeCount();
}

// Search
document.getElementById('searchBox').addEventListener('input', function(e) {
  const query = e.target.value.trim().toLowerCase();
  const main = document.querySelector('.main');

  if (!query) {
    main.classList.remove('filter-mode');
    document.querySelectorAll('.setting, .subsection, .section').forEach(el => {
      el.classList.remove('filter-hidden', 'filter-empty');
    });
    switchSection(activeSection);
    return;
  }

  main.classList.add('filter-mode');

  // Filter settings
  document.querySelectorAll('.section').forEach(sectionEl => {
    let sectionHasMatch = false;

    sectionEl.querySelectorAll('.setting').forEach(el => {
      const match = el.dataset.search.includes(query);
      el.classList.toggle('filter-hidden', !match);
      if (match) sectionHasMatch = true;
    });

    // Hide subsections with no visible settings
    sectionEl.querySelectorAll('.subsection').forEach(sub => {
      const hasVisible = sub.querySelector('.setting:not(.filter-hidden)');
      sub.classList.toggle('filter-hidden', !hasVisible);
    });

    sectionEl.classList.toggle('filter-empty', !sectionHasMatch);
  });

  // Update search results label in sidebar
  const label = document.getElementById('sidebarSearchLabel');
  if (label) {
    const matchCount = document.querySelectorAll('.setting:not(.filter-hidden)').length;
    label.textContent = matchCount > 0 ? 'Search Results (' + matchCount + ')' : 'No results';
  }
});

// Keyboard nav
document.addEventListener('keydown', function(e) {
  if (e.key === 'f' && (e.ctrlKey || e.metaKey)) {
    e.preventDefault();
    document.getElementById('searchBox').focus();
  }
  if (e.key === 'Escape') {
    const search = document.getElementById('searchBox');
    if (document.activeElement === search) {
      search.value = '';
      search.dispatchEvent(new Event('input'));
      search.blur();
    }
  }
});
